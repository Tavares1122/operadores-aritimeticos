# operadores-aritimeticos
# aprendizagem 

n1 = int(input('digite um numero'))
n2 = int(input('digite outro valor'))
s = n1 + n2
m = n1 * n2
di = n1 // n2
p = n1 ** n2
d = n1 / n2
r = n1 % n2
print(
    'a soma é {0} \no produto é {1} \na divisao inteira é {2} \na potencia é {3} \na divisao é {4} \n'' o resto é{5}'.format(
        s, m, di, p, d, r))
